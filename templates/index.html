'PUT' : 'POST'; fetch(url, { method: method, headers: { 'Content-Type': 'application/json', }, body: JSON.stringify(recipeData) }) .then(response => response.json()) .then(res => { hideSaveProgress(); showNotification(res.message, 'success'); resetForm(); loadRecipes(); switchTab("browse"); }) .catch(error => { hideSaveProgress(); showNotification("操作失敗: " + error.message, 'error'); }); }
